\section*{Integration}

\qs{Evaluate $ \int x \sqrt{x^2 + 1} dx $}{
  $$ \int x \sqrt{x^2 + 1} dx $$
  % explain the substitution
  Let $ u = x^2 + 1 $, then $ du = 2x dx $, so $ \frac{1}{2} du = x dx $.
  % substitute
  $$ = \frac{1}{2} \int \sqrt{u} du $$
  % integrate
  $$ = \frac{1}{2} \frac{2}{3} u^{3/2} + C $$ 
  % substitute back
  $$ = \frac{1}{3} (x^2 + 1)^{3/2} + C $$
}

\qs{Evaluate $ \int \frac{\sin x}{\cos x} dx $}{
  $$ \int \frac{\sin x}{\cos x} dx $$ 
  Let $ u = \cos x $ so $ du = -\sin x dx $. 
  $$ = - \int \frac{1}{u} du $$ 
  $$ = - \ln |u| + C $$ 
  $$ = - \ln |\cos x| + C $$
}

\qs{Evaluate $ \int  xe^{3x} dx $}{
  $$ \int  xe^{3x} dx $$
  $$ = \left ( x \frac{e^{3x}}{3} - \int \frac{e^{3x}}{3} dx \right) + C $$
  $$ = \left ( x \frac{e^{3x}}{3} - \frac{1}{9} e^{3x} \right) + C $$
  $$ = \left ( \frac{3x - 1}{9} e^{3x} \right) + C $$
}

\qs{Given $ \int u^n \ln u du = \frac{u^{n+1} \ln u}{n+1} - \frac{u^{n+1}}{(n+1)^2} + C $, evaluate $ \int x^2 \ln (2x) dx $}{
  $$ \int x^2 \ln (2x) dx $$
  Let $ u = 2x $, so $ du = 2 dx $, and $ \frac{1}{2} du = dx $. 
  $$ = \frac{1}{2} \int \frac{1}{4} u^2 \ln u du $$
  $$ = \frac{1}{8} \int u^2 \ln u du $$ 
  $$ = \frac{1}{8} \left( \frac{u^3 \ln u}{3} - \frac{u^3}{9} \right) + C $$
  $$ = \frac{1}{72} \left ( 3 (2x)^3 \ln (2x) - (2x)^3 \right) + C $$
}

\section*{Basic Integration Formulas}

\qs{$ \int \sin (ax) dx $}{
  $$ \int \sin (ax) dx = -\frac{1}{a} \cos (ax) + C $$ 

  a. $ \int \sin  16x dx $ \\
  $ = -\frac{1}{16} \cos 16x + C $ 

  b. $ \int \sin \frac{1}{2}x dx $ \\
  $ = -2 \cos \frac{1}{2}x + C $ 
}

\qs{$ \int \tan (ax) dx $}{
  $$ \int  \tan (ax) dx = -\frac{1}{a} \ln |\cos (ax)| + C $$ 

  a. $ \int  \tan 3x dx $ \\
  $ -\frac{1}{3} \ln \cos 3x + C $ 

  b. $ \int \tan \frac{1}{3} x dx $ \\
  $ -3 \ln |\cos \frac{1}{3} x| + C $
}

\section*{U-substitution}

\qs{$ \int \frac{1}{\theta ^2} \cos \frac{1}{\theta } d\theta $}{
  Let $ u = \frac{1}{\theta } $, so $ du = -\frac{1}{\theta ^2} d\theta $. 
  $$ = -\int \cos u du $$ 
  $$ = -\sin u + C $$ 
}

\qs{$ \int \frac{\sin x}{\cos ^2 x} dx $}{
  Let $ u = \cos x $ so $ du = -\sin x dx $. 
  $$ - \int \frac{1}{u^2} du $$ 
  $$ = \frac{1}{u} + C $$ 
  $$ = \frac{1}{\cos x} + C $$
}

\qs{$ \int  \tan ^4 x \sec ^2 x dx $}{
  Let $ u = \tan x $ so $ du = \sec ^2 x dx $. 
  $$ = \int u^4 du $$ 
  $$ = \frac{u^5}{5} + C $$ 
  $$ = \frac{\tan ^5 x}{5} + C $$
}

\section*{Integration by Parts}

\qs{$ \int t \ln (t+1) dt $}{
  Let $ u = \ln (t+1) $ so $ du = \frac{1}{t+1} dt $. 
  Let $ dv = t dt $ so $ v = \frac{1}{2} t^2 $. 
  $$ = \frac{t^2}{2} \ln (t+1) - \int \frac{1}{t+1} \frac{1}{2} t^2 dt $$
  $$ = \frac{t^2}{2} \ln (t+1) - \frac{1}{2} \int \frac{t^2}{t+1} dt $$
  $$ \frac{t^2}{t+1} = (t-1) - \frac{1}{t+1} $$ 
  $$ = \frac{t^2}{2} \ln (t+1) - \frac{1}{2} \int (t-1) dt - \frac{1}{2} \int \frac{1}{t+1} dt $$ 
  $$ = \frac{t^2}{2} \ln (t+1) - \frac{t^2}{4} + \frac{t}{2} - \frac{1}{2} \ln |t+1| + C $$ 
}

\qs{$ \int \cos ^{-1} x dx$}{
  $$ u = \cos ^{-1} x $$ 
  $$ du = -\frac{1}{\sqrt{1-x^2}} dx $$ 
  $$ dv = dx $$ 
  $$ v = x $$ 
  $$ = x \cos ^{-1} x - \int x (-\frac{1}{\sqrt{1-x^2}}) dx $$ 
  Let $ w = 1-x^2 $, so $ dw = -2x dx $, and $ -\frac{1}{2} dw = x dx $. 
  $$ = x \cos ^{-1} x - \frac{1}{2} \int \frac{1}{\sqrt{w}} dw $$ 
  $$ = x \cos ^{-1} x - \sqrt{w} + C $$ 
  $$ = x \cos ^{-1} x - \sqrt{1-x^2} + C $$
}

\qs{$ \int e^{2x} \sin x dx$}{
  $$ u = e^{2x} $$ 
  $$ du = 2e^{2x} dx $$ 
  $$ dv = \sin x dx $$ 
  $$ v = -\cos x $$ 
  $$ = -e^{2x} \cos x - \int -\cos x 2e^{2x} dx $$ 
  $$ = -e^{2x} \cos x + \int 2e^{2x} \cos x dx $$ 
  $$ u = 2e^{2x} $$ 
  $$ du = 4e^{2x} dx $$ 
  $$ dv = \cos x dx $$ 
  $$ v = \sin x $$ 
  $$ = -e^{2x} \cos x + 2e^{2x} \sin x - \int 4e^{2x} \sin x dx $$  
  $$ 5 \int e^{2x} \sin x dx = -e^{2x} \cos x + 2e^{2x} \sin x $$ 
  $$ \int e^{2x} \sin x dx = \frac{-e^{2x} \cos x + 2e^{2x} \sin x}{5} + C $$ 
}
